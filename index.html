First Component


import React, {
  AppRegistry, Component, Text, View,
} from 'react-native';

class Hello extends Component {
  render() {
    return (
    	<View><Text>Hello, World!</Text></View>
    )
  }	
}	

AppRegistry.registerComponent('Hello', () => Hello);

    borderRadius: 4,

  },

    fontSize: 19,
    fontWeight: 'bold',
  },
  activeTitle: {
